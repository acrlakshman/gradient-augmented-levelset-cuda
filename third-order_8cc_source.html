44 template <typename T, typename T_GRID>
45 void GALS::CPU::ThirdOrder<T, T_GRID>::compute(const Array<T_GRID, T> &alpha, Array<T_GRID, Vec3<T>> &grad_alpha)
46 {
47  const Vec3<int> num_cells = alpha.numCells();
48  const T_GRID &grid = alpha.grid();
49  const Vec3<typename T_GRID::value_type> dx = grid.dX();
50  const auto &axis_vectors = GALS::CPU::Grid<typename T_GRID::value_type, T_GRID::dim>::axis_vectors;
51 
52  for (int i = 0; i < num_cells[0]; ++i)
53  for (int j = 0; j < num_cells[1]; ++j)
54  for (int k = 0; k < num_cells[2]; ++k) {
55  for (int axis = 0; axis < T_GRID::dim; ++axis) {
56  typename T_GRID::value_type one_by_dx = static_cast<typename T_GRID::value_type>(1.) / (6.0 * dx[axis]);
57 
58  grad_alpha(i, j, k)[axis] =
59  (2 * alpha(i + axis_vectors(axis, 0), j + axis_vectors(axis, 1), k + axis_vectors(axis, 2)) +
60  3 * alpha(i, j, k) -
61  6 * alpha(i - axis_vectors(axis, 0), j - axis_vectors(axis, 1), k - axis_vectors(axis, 2)) +
62  1 * alpha(i - 2 * axis_vectors(axis, 0), j - 2 * axis_vectors(axis, 1), k - 2 * axis_vectors(axis, 2))) *
63  one_by_dx;
64  }
65  }
66 }
