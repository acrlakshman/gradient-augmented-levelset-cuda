48 template <typename T, typename T_GRID>
49 void GALS::TEMPORAL_SCHEMES::SEMI_LAGRANGIAN::Euler<T, T_GRID>::compute(
50  const T dt, const GALS::CPU::LevelsetVelocity<T_GRID, T> &levelset_velocity,
51  GALS::CPU::Levelset<T_GRID, T> &levelset)
52 {
53  const auto &grid = levelset.grid();
54  const auto &velocity = levelset_velocity.velocity();
55  const auto &velocity_grad = levelset_velocity.velocityGradient();
56  const GALS::CPU::Vec3<int> num_cells = grid.numCells();
57  auto &phi = levelset.phi();
58  auto &psi = levelset.psi();
59 
60  // Compute x_root `root of the characteristic`.
61  GALS::CPU::Array<T_GRID, GALS::CPU::Vec3<T>> x_root(grid);
62 
63  for (int i = 0; i < num_cells[0]; ++i)
64  for (int j = 0; j < num_cells[1]; ++j)
65  for (int k = 0; k < num_cells[2]; ++k) x_root(i, j, k) = grid(i, j, k) - velocity(i, j, k) * dt;
66 
67  // Compute phi_interp_prev, psi_interp_prev at x_root.
68  GALS::CPU::Interpolate<T, T_GRID, GALS::INTERPOLATION::Hermite<T, T_GRID>>::compute(x_root, levelset);
69 
70  // Compute x_root_grad.
71  GALS::CPU::Array<T_GRID, GALS::CPU::Mat3<T>> x_root_grad(grid);
72 
73  const auto &identity_mat = T_GRID::identity_mat;
74 
75  for (int i = 0; i < num_cells[0]; ++i)
76  for (int j = 0; j < num_cells[1]; ++j)
77  for (int k = 0; k < num_cells[2]; ++k) x_root_grad(i, j, k) = identity_mat - velocity_grad(i, j, k) * dt;
78 
79  const auto &phi_interp_prev = levelset.phiInterpPrev();
80  const auto &psi_interp_prev = levelset.psiInterpPrev();
81 
82  // Update phi and psi.
83  for (int i = 0; i < num_cells[0]; ++i)
84  for (int j = 0; j < num_cells[1]; ++j)
85  for (int k = 0; k < num_cells[2]; ++k) {
86  phi(i, j, k) = phi_interp_prev(i, j, k);
87  psi(i, j, k) = x_root_grad(i, j, k).dot(psi_interp_prev(i, j, k));
88  }
89 }
